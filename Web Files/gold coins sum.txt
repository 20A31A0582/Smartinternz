n=10
k=15
l=[5,3,17,14,18,1,18,4,8,3]
def gold(n,k,l):
    start,end,cs=0,0,0
    while end<n:
        cs+=l[end]
        while cs>k:
            cs-=l[start]
            start+=1 
        if cs==k:
            print(start+1,end+1)
            break
        end+=1 
gold(n,k,l)
